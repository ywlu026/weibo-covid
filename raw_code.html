Word Frequency

# line tokenization
weibo_line <- newdata %>%
  unnest_tokens(
    output = line,
    input = ,
    token = function (x)
      str_split(x, CHUNK_DELIMITER)
  ) %>%
  group_by(index) %>%
  mutate(line_id = row_number()) %>%
  ungroup

# word tokenization
weibo_word <- weibo_line %>%
  unnest_tokens(
    output = word,
    input = line,
    token = function(x)
      segment(x,jiebar = my_seg)
  ) %>%
  group_by(index) %>%
  mutate(word_id = row_number()) %>%
  ungroup
